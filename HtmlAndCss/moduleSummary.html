# 单个导出和倒入

// 导出方法

  console.log(dogName)
  return dogName

# export default

  • 默认导出,每个js文件只能有一个default导出
  • 可以和 export 单个导出共存在一个文件中
// 默认导出函数
export default ()=>{
  console.log('husky')
}

# require引入es6的模块导出

  • 在a.js文件中分别用es6的单个导出和默认导出

export function show(){
  console.log('husky')
}
export function speak(){
  console.log('woof woof')
}
export default like(){
  console.log('food')
}
// index.js
let a = require('./a.js')
